# Set seed for reproducibility
set.seed(385)

# Recipe 
rf_rec <- recipe(pmFRM ~ ., data = train_data) |>
  update_role(date, new_role = "id") |>  # Save date for plotting
  step_mutate(
    yday = lubridate::yday(date),
    month = lubridate::month(date)
  ) |>
  step_corr(all_numeric_predictors(), threshold = 0.9) |>  # Remove highly correlated predictors
  step_nzv(all_predictors())  # Remove near-zero variance predictors

# Model 
rf_model <- rand_forest(
  mtry = tune(),
  min_n = tune()
) |>
  set_engine("ranger", num.threads = parallel::detectCores()) |>
  set_mode("regression")

# Workflow 
rf_wf <- workflow() |>
  add_recipe(rf_rec) |>
  add_model(rf_model)

# Cross-validation setup
folds <- vfold_cv(train_data, v = 10)

# Tuning Grid
rf_grid <- expand_grid(
  mtry = c(1, 2, 3),
  #min_n = c(5, 10, 15)

The min_n parameter sets the minimum number of observations required in a node, with larger values creating a more generalized model.
